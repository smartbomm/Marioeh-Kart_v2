#include <Arduino.h>
#include <BarcodeReader.h>
#include <OdometerData.h>
#include <FlashStorage.h>


barcodeConfig_t config;

uint32_t speed;
uint8_t barcodeValue;
odometerData_t odometerData;
char *string;

struct map_config_t
{
  uint32_t token;
  uint8_t version;
} *map_config_local, map_config_remote;


void setup()
{
  map_config_remote.version = 0x34;

  map_config_local = (map_config_t*) flash_start();
  
  flash_init();

  config.pin = 2u;
  config.bitLength = 7;
  barcode_init(config);
  Serial.begin(9600);
  delay(1000);
  if (map_config_remote.version != map_config_local->version)
  {
    flash_write(&map_config_remote, sizeof(map_config_t));
    Serial.println("Noch nicth vorhanden");
  }
  else
  {
    Serial.println("Vorhanden");
  }
    for (;;);
}

void loop()
{
  barcode_error_t success = barcode_get(&barcodeValue, &speed);
  if (READING_SUCCESSFUL == success)
  {
    char str[100];
    sprintf(str, "Barcode read:\n----------------\nValue: %d\nSpeed [mm/s]: %d\n\n", barcodeValue, speed);
    Serial.print(str);
  }
  else Serial.println(success);
  delay(1000);
}
